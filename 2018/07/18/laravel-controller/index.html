laravel 控制器

创建一个控制器

php artisan make:controller [name]

控制器使用方法

Route::get('user/{id}', 'ControllerURI@methodName')

单动作控制器

__invoke()方法,注册路由时无需指定方法,会被自动调用

控制器中间件

//__construct()构造函数
public function __construct() {
    $this->middleware('token');
}
//指定中间件生效方法
$this->middleware('token')->only('methodName');
$this->middleware('token')->except('methodName');
//指定多个方法
$this->middleware('token')->only(['name1','name2',...]):

资源控制器

//创建一个资源控制器
php artisan make:controller [name] --resource
//注册资源路由
Route::resource('posts', 'ControllerURI');

//一次注册多个资源路由
Route::resources([
    'posts1' => 'URI1',
    'posts2' => 'URI2'
]);

//指定资源路由处理的动作子集
Route::resource('posts', 'ControllerURI', ['only' =>
    ['create', 'index', 'store', 'show', 'edit', 'update', 'destroy']
]);

//快速生成API资源控制器(不含create,edit方法)
php artisan make:controller API/ControllerName --api
//API资源路由
Route::apiResource('posts', 'ControllerURI');

路由缓存

//生成路由缓存
php artisan route:cache
//移除路由缓存文件
php artisan route:clear
